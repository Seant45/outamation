# Different data types
a = 10          # Integer
b = 20.5        # Float
c = "Python"    # String
d = True        # Boolean
print(type(a))  # <class 'int'>
print(type(b))  # <class 'float'>
print(type(c))  # <class 'str'>
print(type(d))  # <class 'bool'>

# Arithmetic Operators
x = 10
y = 5
print(x + y)  # 15
print(x - y)  # 5
print(x * y)  # 50
print(x / y)  # 2.0

# Logical Operators
x = True
y = False
print(x and y)  # False
print(x or y)   # True
print(not x)    # False
